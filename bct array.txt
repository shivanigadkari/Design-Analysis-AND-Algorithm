pragma solidity ^0.8.0;

contract StudentMarks {

    // Dynamic array to store marks
    uint[] public marks;

    // Event to log addition of marks
    event MarksAdded(uint[] newMarks);

    // Function to accept an array of marks
    function addMarks(uint[] memory newMarks) public {
        for (uint i = 0; i < newMarks.length; i++) {
            marks.push(newMarks[i]);
        }
        emit MarksAdded(newMarks);
    }

    // Function to get all marks
    function getMarks() public view returns (uint[] memory) {
        return marks;
    }

    // Function to calculate sum of marks
    function getTotalMarks() public view returns (uint) {
        uint total = 0;
        for (uint i = 0; i < marks.length; i++) {
            total += marks[i];
        }
        return total;
    }

    // Fallback function (optional)
    fallback() external payable {}
    
    // Receive function (optional)
    receive() external payable {}
}
